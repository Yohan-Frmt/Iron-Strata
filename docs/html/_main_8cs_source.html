186 private void DrawCard()
187 {
188 _world.Query<ResourceComponent>()
189 .FirstOptional()
190 .Bind(e => _world.GetOptional<ResourceComponent>(e))
191 .Match(resources =>
192 {
193 if (resources.Scrap < ResourceRegistry.CardDrawCost || _handContainer.GetChildCount() >= 5) return;
194
195 resources.Scrap -= ResourceRegistry.CardDrawCost;
196 var newCard = _cardScene.Instantiate<Scripts.UI.CardUi>();
197 var randomType = GD.Randf() > 0.5f ? WagonType.Combat : WagonType.Storage;
198 _handContainer.AddChild(newCard);
199 newCard.Setup(randomType);
200 }, () => { });
201 }

228 private static Entity CreateWagon(World world, int slot, int layer, WagonType type, string blueprint, float health, Color tint, string label)
229 {
230 var entity = world.CreateEntity();
231 world.Add(entity, new WagonTypeComponent { Type = type, BlueprintId = blueprint });
232 world.Add(entity, new WagonSlotComponent { SlotIndex = slot, Layer = layer });
233 world.Add(entity, new HealthComponent { Max = health, Current = health });
234 world.Add(entity, new RenderableComponent { Tint = tint, Label = label });
235
236 if (type == WagonType.Combat)
237 world.Add(entity, new TurretComponent { Range = 30f, Damage = 15f, FireRate = 10f });
238
239 return entity;
240 }
