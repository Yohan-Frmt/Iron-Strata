33

41 currentLayerNodes.Add(node);
42 }
43
44 map.Add(currentLayerNodes);
45 }
46
47 // 2. Connect nodes between adjacent layers.
48 for (var layer = 0; layer < TotalLayers - 1; layer++)
49 {
50 var currentLayer = map[layer];
51 var nextLayer = map[layer + 1];
52
53 // Ensure every node in the current layer has at least one outgoing connection.
54 foreach (var node in currentLayer)
55 {
56 var target = nextLayer[GD.RandRange(0, nextLayer.Count - 1)];
57 node.NextNodes.Add(target.Id);
58 }
59
60 // Ensure every node in the next layer has at least one incoming connection.
61 foreach (var nextNode in nextLayer)
62 {
63 if (currentLayer.Any(n => n.NextNodes.Contains(nextNode.Id))) continue;
64
65 var source = currentLayer[GD.RandRange(0, currentLayer.Count - 1)];
66 if (!source.NextNodes.Contains(nextNode.Id))
67 {
68 source.NextNodes.Add(nextNode.Id);
69 }
70 }
71 }
72
