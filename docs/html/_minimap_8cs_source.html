94 private static Vector2 GetTrainMapPosition(MapComponent map, LocationComponent loc)

99 var dist = start.DistanceTo(end);
100 var t = dist > 0 ? Mathf.Clamp(loc.TravelProgress / dist, 0, 1) : 0;
101
102 return start.Lerp(end, t);
