28 public void Update(World world, double delta)
29 {
30 var isInCity = world.Query<LocationComponent>()
31 .FirstOptional()
32 .Bind(e => world.GetOptional<LocationComponent>(e))
33 .Match(l => l.IsInCityZone, () => false);
34
35 if (isInCity)
36 {
37 _hordeTimer += (float)delta;
38 if (_hordeTimer >= HordeSpawnInterval)
39 {
40 _hordeTimer = 0f;
41 foreach (var rule in EnemyRegistry.SpawnRules.Where(rule => GD.Randf() <= rule.Chance)) SpawnHorde(world, rule.Count, rule.Type);
42 }
43 }
44 else
45 {
46 _hordeTimer = 0f;
47 }
48
49 var wagons = world.Query<WagonSlotComponent, WagonTypeComponent, HealthComponent>().ToList();
50 if (wagons.Count == 0) return;
51 var allEnemies = world.Query<EnemyComponent, PositionComponent>().ToList();
52 foreach (var entity in world.Query<EnemyComponent, PositionComponent, MovementComponent>())
53 {
54 // Spatially handle enemy movement and interactions.
55 var enemy = world.Get<EnemyComponent>(entity);
56 var pos = world.Get<PositionComponent>(entity);
57 var loco = world.Get<MovementComponent>(entity);
58
59 // Target acquisition.
60 var needsNewTarget = enemy.CurrentTarget.Match(target => !world.IsAlive(target), () => true);
61 if (needsNewTarget)
62 enemy.CurrentTarget = FindBestTarget(enemy.Type, wagons, world);
63
64 enemy.CurrentTarget.Match(target =>
65 {
66 var slotComp = world.Get<WagonSlotComponent>(target);
67 const float wagonSize = 5f;
68 const float wagonPhysicalRadius = 4.0f; // Collision radius of the wagon
69
70 // Calculate target position in global space.
71 var targetGlobalPos = trainRoot.GlobalPosition + new Vector3(-slotComp.SlotIndex * wagonSize, 2.0f, 0);
72 var distanceToCenter = pos.Value.DistanceTo(targetGlobalPos);
73
74 // Combat logic.
75 if (distanceToCenter <= enemy.AttackRange + wagonPhysicalRadius)
76 {
77 enemy.AttackTimer -= (float)delta;
78 if (enemy.AttackTimer <= 0)
79 {
80 enemy.AttackTimer = 1f / enemy.AttackSpeed;
81 var health = world.Get<HealthComponent>(target);
82 health.Current -= enemy.Damage;
83
84 if (health.Current <= 0) HandleWagonDestruction(world, target);
85 }
86 }
87 else
88 {
89 // Movement and Boids-like behavior (separation).
90 var moveDir = (targetGlobalPos - pos.Value).Normalized();
91 var repulsionForce = Vector3.Zero;
92 var neighbors = 0;
93
94 // Repulsion from other enemies.
95 foreach (var otherEntity in allEnemies)
96 {
97 if (entity.Equals(otherEntity)) continue;
98
99 var otherPos = world.Get<PositionComponent>(otherEntity).Value;
100 var distToOther = pos.Value.DistanceTo(otherPos);
101 const float separationRadius = 10.0f;
102
103 if (!(distToOther < separationRadius) || !(distToOther > 0.01f)) continue;
104
105 var pushDir = (pos.Value - otherPos).Normalized();
106 repulsionForce += pushDir * (separationRadius - distToOther);
107 neighbors++;
108 }
109
110 // Repulsion from wagons to avoid clipping.
111 foreach (var wagon in wagons)
112 {
113 var wSlot = world.Get<WagonSlotComponent>(wagon);
114 var wPos = trainRoot.GlobalPosition + new Vector3(-wSlot.SlotIndex * wagonSize, 2.0f, 0);
115 var distToWagon = pos.Value.DistanceTo(wPos);
116
117 if (!(distToWagon < wagonPhysicalRadius + 0.5f)) continue;
118
119 var pushDir = (pos.Value - wPos).Normalized();
120 repulsionForce += pushDir * ((wagonPhysicalRadius + 0.5f) - distToWagon) * 5.0f;
121 neighbors++;
122 }
123
124 if (neighbors > 0)
125 {
126 repulsionForce /= neighbors;
127 moveDir = (moveDir + repulsionForce * 1.5f).Normalized();
128 }
129
130 // Apply movement.
131 pos.Value += moveDir * loco.Speed * (float)delta;
132
133 // Simple gravity for non-flying enemies.
134 if (!loco.IsFlying)
135 {
136 pos.Value.Y -= 9.81f * (float)delta;
137 if (pos.Value.Y < 0) pos.Value.Y = 0; // Ground level
138 }
139 }
140 }, () => { });
141 }

148 private void HandleWagonDestruction(World world, Entity target)
149 {
150 var hitSlot = world.Get<WagonSlotComponent>(target);
151 GD.Print($"Wagon at index {hitSlot.SlotIndex} (layer {hitSlot.Layer}) destroyed!");
152
153 if (hitSlot.SlotIndex == 0) GD.PrintErr("!!! GAME OVER - LOCOMOTIVE DESTROYED !!!");
154
155 var allWagons = world.Query<WagonSlotComponent>().ToList();
156 foreach (var w in allWagons)
157 {
158 if (!world.IsAlive(w)) continue;
159 var wSlot = world.Get<WagonSlotComponent>(w);
160
161 // Destroy this wagon if it's the one hit OR if it's stacked on top of it.
162 if (wSlot.SlotIndex == hitSlot.SlotIndex && wSlot.Layer >= hitSlot.Layer)
163 {
164 DestroyWagon(world, w);
165 }
166 // If it's further back in the train, it becomes detached and abandoned.
167 else if (wSlot.SlotIndex > hitSlot.SlotIndex)
168 {
169 AbandonWagon(world, w, trainRoot);
170 }
171 }

189 private static void AbandonWagon(World world, Entity e, Node3D nodeRoot)
190 {
191 if (world.Has<RenderableComponent>(e))
192 {
193 var node = world.Get<RenderableComponent>(e).Node;
194 if (node != null)
195 {
196 var globalTrans = node.GlobalTransform;
197 // Detach from train and add to the main scene.
198 node.GetParent()?.RemoveChild(node);
199 nodeRoot.GetTree().CurrentScene.AddChild(node);
200 node.GlobalTransform = globalTrans;
201
202 // Cleanup after a few seconds to avoid clutter.
203 var timer = node.GetTree().CreateTimer(6.0f);
204 timer.Timeout += () =>
205 {
206 if (GodotObject.IsInstanceValid(node)) node.QueueFree();
207 };
208 }
209 }

217 private static Option<Entity> FindBestTarget(EnemyType enemyType, List<Entity> wagons, World world)
218 {
219 var bestTarget = Option<Entity>.None;
220 var bestScore = -99999f;
221
222 foreach (var wagon in wagons)
223 {
224 var typeComp = world.Get<WagonTypeComponent>(wagon);
225 var slotComp = world.Get<WagonSlotComponent>(wagon);
226
227 // Base score favors higher wagons and certain types.
228 var score = GetDefaultTypePriority(typeComp.Type) + (slotComp.Layer * 50);
229
230 // Type-specific targeting biases.
231 switch (enemyType)
232 {
233 case EnemyType.Safeguard:
234 // Safeguards focus heavily on vertical structures.
235 score += slotComp.Layer * 10000;
236 break;
237 case EnemyType.Wasp:
238 // Wasps prioritize combat wagons to disable defenses.
239 switch (typeComp.Type)
240 {
241 case WagonType.Combat:
242 score += 5000;
243 break;
244 }
245 break;
246 case EnemyType.Crawler:
247 // Crawlers focus on living and research wagons.
248 switch (typeComp.Type)
249 {
250 case WagonType.Living:
251 score += 5000;
252 break;
253 case WagonType.Research:
254 score += 4000;
255 break;
256 }
257 break;
258 }
259
260 if (!(score > bestScore)) continue;
261 bestScore = score;
262 bestTarget = Option<Entity>.Some(wagon);
263 }
264
265 return bestTarget;
266 }

283 private void SpawnHorde(World world, int count, EnemyType type)
284 {
285 var def = EnemyRegistry.EnemyDefs[type];
286 var angle = GD.Randf() * Mathf.Tau;
287 const float distance = 80f;
288
289 // Pick a point on a circle around the train.
290 var hordeEpicenter = trainRoot.GlobalPosition +
291 new Vector3(Mathf.Cos(angle) * distance, 0, Mathf.Sin(angle) * distance);
292
293 for (var i = 0; i < count; i++)
294 {
295 var e = world.CreateEntity();
296
297 // Apply dispersion so they don't spawn on top of each other.
298 var randomOffset = new Vector3(
299 (GD.Randf() - 0.5f) * def.DispersionRadius * 2f,
300 def.IsFlying ? 5.0f : 0f,
301 (GD.Randf() - 0.5f) * def.DispersionRadius * 2f
302 );
303
304 world.Add(e, new PositionComponent { Value = hordeEpicenter + randomOffset });
305 world.Add(e, new MovementComponent { Speed = def.Speed + (GD.Randf() * 2f), IsFlying = def.IsFlying });
306 world.Add(e, new HealthComponent { Max = def.Health, Current = def.Health });
307 world.Add(e,
308 new EnemyComponent
309 {
310 Type = def.Type,
311 Damage = def.Damage,
312 AttackRange = def.AttackRange,
313 AttackSpeed = def.AttackSpeed
314 });
315 }
