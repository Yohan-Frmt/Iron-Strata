68 public void UpdatePreview(World world, WagonType cardType, Vector2 mousePos)
69 {
70 var locOption = world.Query<LocationComponent>()
71 .FirstOptional()
72 .Bind(e => world.GetOptional<LocationComponent>(e));
73
74 if (locOption.Match(l => !l.IsInTransit, () => true)) return;
75
76 // Perform raycast from camera to world.
77 var spaceState = _bottomHud.GetViewport().World3D.DirectSpaceState;
78 var rayOrigin = _camera.ProjectRayOrigin(mousePos);
79 var rayEnd = rayOrigin + _camera.ProjectRayNormal(mousePos) * 1000f;
80 var query = PhysicsRayQueryParameters3D.Create(rayOrigin, rayEnd);
81 query.CollideWithAreas = true;
82
83 var result = spaceState.IntersectRay(query);
84
85 if (result.Count <= 0) return;
86
87 var collider = (Node3D)result["collider"];
88
89 // Check if we hit an existing wagon.
90 if (collider.HasMeta("EntityId"))
91 {
92 var entityId = (int)collider.GetMeta("EntityId");
93 var hitEntity = new Entity(entityId);
94
95 if (world.IsAlive(hitEntity))
96 {
97 var hitSlot = world.Get<WagonSlotComponent>(hitEntity);
98 var highestLayer = -1;
99 var topType = WagonType.Locomotive;
100
101 // Find the topmost wagon at this slot index.
102 foreach (var e in world.Query<WagonSlotComponent, WagonTypeComponent>())
103 {
104 var s = world.Get<WagonSlotComponent>(e);
105 if (s.SlotIndex != hitSlot.SlotIndex || s.Layer <= highestLayer) continue;
106 highestLayer = s.Layer;
107 topType = world.Get<WagonTypeComponent>(e).Type;
108 }
109
110 // Cannot build on top of the locomotive.
111 if (topType == WagonType.Locomotive)
112 {
113 _previewGhost.Visible = false;
114 return;
115 }
116
117 // If types match, it's an upgrade (yellow highlight).
118 if (topType == cardType)
119 {
120 _previewMat.AlbedoColor = new Color(1.0f, 0.8f, 0.2f, 0.6f);
121 _previewMat.Emission = new Color(1.0f, 0.8f, 0.2f);
122 _previewGhost.Position = TrainLayout.GetLocalPosition(hitSlot.SlotIndex, highestLayer);
123 _previewGhost.Scale = new Vector3(1.1f, 1.1f, 1.1f);
124 }
125 // Otherwise, it's a new layer (blue highlight).
126 else
127 {
128 _previewMat.AlbedoColor = new Color(0.2f, 0.6f, 1.0f, 0.6f);
129 _previewMat.Emission = new Color(0.2f, 0.6f, 1.0f);
130 _previewGhost.Position = TrainLayout.GetLocalPosition(hitSlot.SlotIndex, highestLayer + 1);
131 _previewGhost.Scale = Vector3.One;
132 }
133 _previewGhost.Visible = true;
134 return;
135 }
136 }
137 // Check if we hit the floor to append a new wagon at the end of the train.
138 else if (collider.Name == "FloorBody")
139 {
140 var hitPoint = (Vector3)result["position"];
141 var localHit = _trainRoot.ToLocal(hitPoint);
142 var maxSlot = world.Query<WagonSlotComponent>()
143 .Select(e => world.Get<WagonSlotComponent>(e).SlotIndex)
144 .Prepend(0).Max();
145
146 var lastWagonX = -maxSlot * (TrainLayout.WagonLength + TrainLayout.WagonGap);
147
148 // If mouse is behind the last wagon, show placement ghost (green highlight).
149 if (localHit.X < lastWagonX + 2f)
150 {
151 _previewMat.AlbedoColor = new Color(0.2f, 1.0f, 0.2f, 0.6f);
152 _previewMat.Emission = new Color(0.2f, 1.0f, 0.2f);
153 _previewGhost.Position = TrainLayout.GetLocalPosition(maxSlot + 1, 0);
154 _previewGhost.Scale = Vector3.One;
155 _previewGhost.Visible = true;
156 return;
157 }
158 }
159 _previewGhost.Visible = false;
160 }

166 public bool TryPlayCard(World world, WagonType cardType, int cost, Vector2 mousePos)
167 {
168 var locOption = world.Query<LocationComponent>()
169 .FirstOptional()
170 .Bind(e => world.GetOptional<LocationComponent>(e));
171
172 if (locOption.Match(l => !l.IsInTransit, () => true)) return false;
173
174 // Check if player has enough resources.
175 var resOption = world.Query<ResourceComponent>()
176 .FirstOptional()
177 .Bind(e => world.GetOptional<ResourceComponent>(e));
178
179 if (resOption.Match(r => r.Scrap < cost, () => true)) return false;
180 var resources = resOption.Unwrap();
181
182 var spaceState = _bottomHud.GetViewport().World3D.DirectSpaceState;
183 var rayOrigin = _camera.ProjectRayOrigin(mousePos);
184 var rayEnd = rayOrigin + _camera.ProjectRayNormal(mousePos) * 1000f;
185 var query = PhysicsRayQueryParameters3D.Create(rayOrigin, rayEnd);
186 query.CollideWithAreas = true;
187 var result = spaceState.IntersectRay(query);
188
189 if (result.Count <= 0) return false;
190 var collider = (Node3D)result["collider"];
191
192 // Handle hitting an existing wagon.
193 if (collider.HasMeta("EntityId"))
194 {
195 var entityId = (int)collider.GetMeta("EntityId");
196 ApplyCardToWagon(world, new Entity(entityId), cardType);
197 resources.Scrap -= cost;
198 return true;
199 }
200
201 // Handle hitting the ground to append a new wagon.
202 if (collider.Name != "FloorBody") return false;
203
204 var hitPoint = (Vector3)result["position"];
205 var localHit = _trainRoot.ToLocal(hitPoint);
206 var maxSlot = world.Query<WagonSlotComponent>()
207 .Select(e => world.Get<WagonSlotComponent>(e).SlotIndex)
208 .Prepend(0).Max();
209
210 var lastWagonX = -maxSlot * (TrainLayout.WagonLength + TrainLayout.WagonGap);
211 if (!(localHit.X < lastWagonX + 2f)) return false;
212
213 CreateNewWagon(world, maxSlot + 1, 0, cardType);
214 resources.Scrap -= cost;
215 return true;
216 }

222 private static void ApplyCardToWagon(World world, Entity hitEntity, WagonType cardType)
223 {
224 if (!world.IsAlive(hitEntity)) return;
225 var hitSlot = world.Get<WagonSlotComponent>(hitEntity);
226 var topEntity = hitEntity;
227 var highestLayer = -1;
228
229 // Find the topmost wagon in the stack.
230 foreach (var e in world.Query<WagonSlotComponent, WagonTypeComponent>())
231 {
232 var s = world.Get<WagonSlotComponent>(e);
233 if (s.SlotIndex != hitSlot.SlotIndex || s.Layer <= highestLayer) continue;
234 highestLayer = s.Layer;
235 topEntity = e;
236 }
237
238 var topType = world.Get<WagonTypeComponent>(topEntity).Type;
239 if (topType == WagonType.Locomotive) return;
240
241 // Upgrade if types match.
242 if (topType == cardType)
243 {
244 var health = world.Get<HealthComponent>(topEntity);
245 health.Max += 50f;
246 health.Current += 50f;
247
248 // Special case for combat wagon upgrades.
249 if (cardType == WagonType.Combat && world.Has<TurretComponent>(topEntity))
250 {
251 var turret = world.Get<TurretComponent>(topEntity);
252 turret.Damage += 10f;
253 turret.FireRate += 3f;
254 }
255 }
256 // Build new layer if types differ.
257 else
258 {
259 CreateNewWagon(world, hitSlot.SlotIndex, highestLayer + 1, cardType);
260 }

279 world.Add(entity, new WagonSlotComponent { SlotIndex = slot, Layer = layer });
280 world.Add(entity, new HealthComponent { Max = 150f, Current = 150f });
281 world.Add(entity, new RenderableComponent { Tint = tint, Label = type.ToString().ToUpper() });
