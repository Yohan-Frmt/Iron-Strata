31 public void Update(World world, double delta)
32 {
33 // Get the total distance traveled for time-based visual effects.
34 var distance = world.Query<TrainMovementComponent>()
35 .FirstOptional()
36 .Bind(e => world.GetOptional<TrainMovementComponent>(e))
37 .Match(mv => mv.DistanceTraveled, () => 0f);
38
39 foreach (var entity in world.Query<RenderableComponent, WagonSlotComponent>())
40 {
41 var r = world.Get<RenderableComponent>(entity);
42 var slot = world.Get<WagonSlotComponent>(entity);
43
44 // Lazy creation of the visual representation if it doesn't exist.
45 if (r.Node == null)
46 {
47 r.Node = BuildSafeWagon(entity, r);
48 _trainRoot.AddChild(r.Node);
49 }
50
51 // Apply a procedural bobbing animation based on distance and slot index.
52 var bobbingPhase = (distance * 0.5f) + (slot.SlotIndex * 1.5f);
53 var bobY = Mathf.Sin(bobbingPhase) * 0.1f;
54
55 var basePos = TrainLayout.GetLocalPosition(slot.SlotIndex, slot.Layer);
56 r.Node.Position = basePos + new Vector3(0, bobY, 0);
57 }
